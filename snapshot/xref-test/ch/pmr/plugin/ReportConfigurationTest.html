56  	@Test
57  	@DisplayName("getModels() returns stream with all bpmn, dmn and cmmn models")
58  	void getModels_allBpmnDmnCmmn() {
59  		File bpmn1 = new File("target/bpmn1");
60  		File bpmn2 = new File("target/bpmn2");
61  		File dmn = new File("target/dmn");
62  		File cmmn1 = new File("target/cmmn1");
63  		File cmmn2 = new File("target/cmmn2");
64  		File cmmn3 = new File("target/cmmn3");
65  		ReportConfiguration testee = new ReportConfiguration(null, Arrays.asList(bpmn1, bpmn2), Arrays.asList(dmn),
66  				Arrays.asList(cmmn1, cmmn2, cmmn3));
67  
68  		Stream<File> models = testee.getModels();
69  
70  		assertThat(models.collect(Collectors.toList()), containsInAnyOrder(bpmn1, bpmn2, dmn, cmmn1, cmmn2, cmmn3));
71  	}
