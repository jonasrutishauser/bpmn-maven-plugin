23  public class ReportMojoIT {
24  
25  	@Rule
26  	public final TestResources resources = new TestResources();
27  
28  	public final MavenRuntime mavenRuntime;
29  
30  	public ReportMojoIT(MavenRuntimeBuilder builder) throws Exception {
31  		this.mavenRuntime = builder.build();
32  	}
33  
34  	@Test
35  	public void buildExample() throws Exception {
36  		File basedir = resources.getBasedir("example");
37  		Path bpmnPath = basedir.toPath().resolve("src/main/resources/diagram.bpmn");
38  		Files.createDirectories(bpmnPath.getParent());
39  		Path dmnPath = basedir.toPath().resolve("src/models/table.dmn");
40  		Files.createDirectories(dmnPath.getParent());
41  		Path cmmnPath = basedir.toPath().resolve("src/diagram.cmmn");
42  		Files.createDirectories(cmmnPath.getParent());
43  		Files.copy(Paths.get("src", "test", "models", "diagram.bpmn"), bpmnPath);
44  		Files.copy(Paths.get("src", "test", "models", "table.dmn"), dmnPath);
45  		Files.copy(Paths.get("src", "test", "models", "diagram.cmmn"), cmmnPath);
46  
47  		MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "site");
48  
49  		result.assertErrorFreeLog();
50  		assertTrue(new File(basedir, "target/site/business-process-models.html").exists());
51  		assertTrue(new File(basedir, "target/site/models/src-main-resources-diagram.bpmn").exists());
52  		assertTrue(new File(basedir, "target/site/models/src-models-table.dmn").exists());
53  		assertTrue(new File(basedir, "target/site/models/src-diagram.cmmn").exists());
54  		assertTrue(new File(basedir, "target/site/bpmn-js").exists());
55  		assertTrue(new File(basedir, "target/site/dmn-js").exists());
56  		assertTrue(new File(basedir, "target/site/cmmn-js").exists());
57  	}
